maid_fill_ideas_slots_effect = {
	if = {
		limit = {
			and = {
				not = {
					has_country_flag = maid_idea_slot_using_1
				}
				not = {
					has_country_flag = maid_idea_slot_using_2
				}
			}
		}
		set_country_flag = maid_idea_slot_using_1
	}
	else_if = {
		limit = {
			has_country_flag = maid_idea_slot_using_1
			not = {
				has_country_flag = maid_idea_slot_using_2
			}
		}
		set_country_flag = maid_idea_slot_using_2
	}
}

maid_remove_ideas_slots_effect = {
	if = {
		limit = {
			has_country_flag = maid_idea_slot_using_1
			not = {
				has_country_flag = maid_idea_slot_using_2
			}
		}
		clr_country_flag = maid_idea_slot_using_1
	}
	else_if = {
		limit = {
			and = {
				has_country_flag = maid_idea_slot_using_1
				has_country_flag = maid_idea_slot_using_2
			}
		}
		clr_country_flag = maid_idea_slot_using_2
	}
}
